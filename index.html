# Two Pointers Pattern
def two_sum_sorted(arr, target):
    left, right = 0, len(arr) - 1
    while left < right:
        current_sum = arr[left] + arr[right]
        if current_sum == target:
            return [left, right]
        elif current_sum < target:
            left += 1
        else:
            right -= 1
# Sliding Window Pattern
def max_subarray_sum(arr, k):
    window_sum = sum(arr[:k])
    max_sum = window_sum
    
    for i in range(k, len(arr)):
        window_sum = window_sum - arr[i-k] + arr[i]
        max_sum = max(max_sum, window_sum)
    
    return max_sum

💡 Study Tips & Best Practices

⏰

Consistency > Intensity

2-3 hours daily beats 12 hours on weekends. Build sustainable habits.

📝

Document Everything

Keep a problem-solving journal. Write down patterns, mistakes, and insights.

🔄

Spaced Repetition

Review problems after 1 day, 1 week, 1 month. Solidifies understanding.

🗣️

Explain Out Loud

Verbalize your approach. If you can't explain it, you don't understand it.
